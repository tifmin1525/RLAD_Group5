b = NULL
for(i in 1:length(CPBLA)){
  m <- gregexpr(pat, CPBLA[[i]][1], perl = T)
  M <- gregexpr(pat, CPBLA[[i]][2], perl = T)
  a[[i]] <- regmatches(CPBLA[[i]][1], m)
  b[[i]] <- regmatches(CPBLA[[i]][2], M)
}
